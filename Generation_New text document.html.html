Calculation Engine (Refactored) --- def get_freq_step(freq): return round(float(freq), 2) def calculate_general_seller_rate(row, rr): """Calculates Rate for General Seller based on Frequency Steps.""" f = get_freq_step(row['Freq']) direction = row['Direction'] # -- Over Injection (Receivable) -- if direction == "Over Injection": rate_slab2 = 0.0 # Beyond VL1 if f >= 50.10: rate_slab2 = rr * -0.10 # Penalty # Within VL1 if f == 50.00: rate_slab1 = rr elif 50.00 < f <= 50.05: steps = (f - 50.00) / 0.01 rate_slab1 = rr * (1 - (0.10 * steps)) elif 49.90 <= f < 50.00: steps = (50.00 - f) / 0.01 rate_slab1 = rr * (1 + (0.015 * steps)) elif 50.05 < f < 50.10: rate_slab1 = 0.0 elif f >= 50.10: rate_slab1 = rr * -0.10 elif f < 49.90: rate_slab1 = rr * 1.15 else: rate_slab1 = rr return rate_slab1, rate_slab2 # -- Under Injection (Payable) -- else: # Beyond VL1 if f >= 50.00: rate_slab2 = rr elif 49.90 <= f < 50.00: rate_slab2 = rr * 1.50 else: rate_slab2 = rr * 2.00 # Within VL1 if f == 50.00: rate_slab1 = rr elif 50.00 < f <= 50.05: steps = (f - 50.00) / 0.01 rate_slab1 = rr * (1 - (0.03 * steps)) elif 49.90 <= f < 50.00: steps = (50.00 - f) / 0.01 rate_slab1 = rr * (1 + (0.05 * steps)) elif f > 50.05: rate_slab1 = rr * 0.85 elif f < 49.90: rate_slab1 = rr * 1.50 else: rate_slab1 = rr return rate_slab1, rate_slab2 def run_dsm_calculation(df): """Core logic applied to a standardized DataFrame""" try: # Common Calculations df['RR'] = df['PPA_Input'] * 10 df['Deviation_MW'] = df['Actual'] - df['Schedule'] df['Abs_Dev_MW'] = df['Deviation_MW'].abs() df['Direction'] = np.where(df['Deviation_MW'] >= 0, 'Over Injection', 'Under Injection') df['Dev_MWh'] = df['Abs_Dev_MW'] * 0.25 # --- A. GENERAL SELLER --- if seller_type == "General Seller": df['VL1_MW'] = (df['Schedule'].abs() * 0.10).clip(upper=100) df['Slab1_MW'] = df[['Abs_Dev_MW', 'VL1_MW']].min(axis=1) df['Slab2_MW'] = (df['Abs_Dev_MW'] - df['VL1_MW']).clip(lower=0) rate_results = df.apply(lambda row: calculate_general_seller_rate(row, row['RR']), axis=1) df['Rate_Slab1'] = [x[0] for x in rate_results] df['Rate_Slab2'] = [x[1] for x in rate_results] df['Charge_Slab1'] = df['Slab1_MW'] * 0.25 * df['Rate_Slab1'] df['Charge_Slab2'] = df['Slab2_MW'] * 0.25 * df['Rate_Slab2'] df['Total_Payable'] = df['Charge_Slab1'] + df['Charge_Slab2'] # --- B. RoR HYDRO SELLER --- elif seller_type == "General Seller (RoR)": df['VL1_MW'] = (df['Schedule'].abs() * 0.10).clip(upper=100) df['VL2_MW'] = (df['Schedule'].abs() * 0.15).clip(upper=150) df['Slab1_MW'] = df[['Abs_Dev_MW', 'VL1_MW']].min(axis=1) rem_after_slab1 = (df['Abs_Dev_MW'] - df['VL1_MW']).clip(lower=0) vl2_band_width = df['VL2_MW'] - df['VL1_MW'] df['Slab2_MW'] = rem_after_slab1.clip(upper=vl2_band_width) df['Slab3_MW'] = (rem_after_slab1 - vl2_band_width).clip(lower=0) def get_ror_rates(row): rr = row['RR'] if row['Direction'] == 'Over Injection': return rr, 0.0, 0.0 else: return rr, rr * 1.05, rr * 1.10 rates = df.apply(get_ror_rates, axis=1, result_type='expand') df[['Rate_Slab1', 'Rate_Slab2', 'Rate_Slab3']] = rates df['Total_Payable'] = (df['Slab1_MW'] * 0.25 * df['Rate_Slab1']) + \ (df['Slab2_MW'] * 0.25 * df['Rate_Slab2']) + \ (df['Slab3_MW'] * 0.25 * df['Rate_Slab3']) # --- C. WS SELLER --- elif seller_type == "WS Seller (Solar/Wind)": l1, l2, l3 = (5, 10, 20) if ws_sub_type == "Solar/Hybrid" else (10, 15, 25) safe_avc = df['AVC'].replace(0, 0.0001) df['VL1_MW'] = (df['AVC'] * l1) / 100 df['VL2_MW'] = (df['AVC'] * l2) / 100 df['VL3_MW'] = (df['AVC'] * l3) / 100 df['Slab1_MW'] = df[['Abs_Dev_MW', 'VL1_MW']].min(axis=1) rem_1 = (df['Abs_Dev_MW'] - df['VL1_MW']).clip(lower=0) band_2 = df['VL2_MW'] - df['VL1_MW'] df['Slab2_MW'] = rem_1.clip(upper=band_2) rem_2 = (rem_1 - band_2).clip(lower=0) band_3 = df['VL3_MW'] - df['VL2_MW'] df['Slab3_MW'] = rem_2.clip(upper=band_3) df['Slab4_MW'] = (rem_2 - band_3).clip(lower=0) def get_ws_rates(row): rr = row['RR'] if row['Direction'] == 'Over Injection': return rr, rr*0.90, rr*0.50, 0.0 else: return rr, rr*1.10, rr*1.50, rr*2.00 rates = df.apply(get_ws_rates, axis=1, result_type='expand') df[['Rate_S1', 'Rate_S2', 'Rate_S3', 'Rate_S4']] = rates df['Total_Payable'] = (df['Slab1_MW'] * 0.25 * df['Rate_S1']) + \ (df['Slab2_MW'] * 0.25 * df['Rate_S2']) + \ (df['Slab3_MW'] * 0.25 * df['Rate_S3']) + \ (df['Slab4_MW'] * 0.25 * df['Rate_S4']) # Final Summary fields df['Paid_at_RR'] = df['Dev_MWh'] * df['RR'] df['Penalty_Loss'] = df['Total_Payable'] - df['Paid_at_RR'] return df except Exception as e: st.error(f"Calculation Error: {e}") return None # --- 3.
